void UTangoMissionComponent::RegisterRandomizationParams(const FRandomStream& InStream,
                                                       const FRandomizationParams& InParams,
                                                       ATangoLevelLayout* InLayout)
{
  // Initialize our local stream from the Randomization Manager's initial seed - we might need it later
  RandomStream.Initialize(InStream.GetInitialSeed()); 
  RandomizationParams = InParams;
  ChosenLevelLayout = InLayout;
  OnRep_ChosenLevelLayout();

  // Only set the objective modes and mission description if any are relevant
  if(InParams.ObjectiveModes.Num() > 0)
  {
    ChosenObjectiveModes = InParams.ObjectiveModes;
    OnRep_ChosenObjectiveModes();
    MissionDescription = GetOwner<ATangoObjectiveManager>()->GetDescriptionFromObjectiveModes(ChosenObjectiveModes);
    OnRep_MissionDescription();
  }
}
