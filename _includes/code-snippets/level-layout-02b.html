void UTangoTakedownMissionComponent::RegisterRandomizationParams(const FRandomStream& InStream,
  const FRandomizationParams& InParams, ATangoLevelLayout* InLayout)
{
  // Initial setup
  Super::RegisterRandomizationParams(InStream, InParams, InLayout);
  TakedownLevelLayout = Cast<ATangoTakedownLevelLayout>(InLayout);
  SetTakedownState(ETakedownMissionState::PreRally);

  // Setup the Rally Zone
  RallyZone = TakedownLevelLayout->RallyZone;
  if(IsValid(RallyZone))
  {
    RallyZone->OnTriggered.AddUniqueDynamic(this, &ThisClass::DelayedRallyZoneTriggered);
  }
  else
  {
    UE_LOG(LogTangoObjectives, Error, TEXT("%s Takedown Level Layout did not have a valid Rally Zone!"),
      *InParams.Layout.Get()->Label.ToString())
  }
}
